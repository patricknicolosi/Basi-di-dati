Trovare la persona che ha il saldo totale più elevato SELECT contomassimo.id_persona FROM (SELECT contocorrente.id_conto, titolare.id_persona, MAX(contocorrente.saldo) FROM contocorrente, titolare WHERE titolare.id_conto = contocorrente.id_conto) as contomassimo,persona WHERE persona.id_persona = contomassimo.id_persona; Test
